197  template <typename T>
198  std::shared_ptr<typename T::Response> call_service_and_wait(
199  rclcpp::Client<T> & client, std::shared_ptr<typename T::Request> request,
200  rclcpp::Executor & service_executor, bool update_controller_while_spinning = false)
201  {
202  EXPECT_TRUE(client.wait_for_service(std::chrono::milliseconds(500)));
203  auto result = client.async_send_request(request);
204  // Wait for the result.
205  if (update_controller_while_spinning)
206  {
207  while (service_executor.spin_until_future_complete(result, std::chrono::milliseconds(50)) !=
208  rclcpp::FutureReturnCode::SUCCESS)
209  {
210  cm_->update(time_, rclcpp::Duration::from_seconds(0.01));
211  }
212  }
213  else
214  {
215  EXPECT_EQ(
216  service_executor.spin_until_future_complete(result), rclcpp::FutureReturnCode::SUCCESS);
217  }
218  return result.get();
219  }
220 
221 protected:
222  rclcpp::TimerBase::SharedPtr update_timer_;
223  std::future<void> executor_spin_future_;
224 };
