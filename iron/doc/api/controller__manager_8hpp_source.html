579  void call_controller_update()
580  {
581  using TimePoint = std::chrono::system_clock::time_point;
582  unsigned int used_update_rate =
583  controller_->get_update_rate() == 0
584  ? cm_update_rate_
585  : controller_
586  ->get_update_rate(); // determines if the controller's or CM's update rate is used
587 
588  while (!terminated_.load(std::memory_order_relaxed))
589  {
590  auto const period = std::chrono::nanoseconds(1'000'000'000 / used_update_rate);
591  TimePoint next_iteration_time =
592  TimePoint(std::chrono::nanoseconds(controller_->get_node()->now().nanoseconds()));
593 
594  if (controller_->get_state().id() == lifecycle_msgs::msg::State::PRIMARY_STATE_ACTIVE)
595  {
596  // critical section, not implemented yet
597  }
598 
599  next_iteration_time += period;
600  std::this_thread::sleep_until(next_iteration_time);
601  }
602  }
603 
604  private:
605  std::atomic<bool> terminated_;
606  std::shared_ptr<controller_interface::ControllerInterfaceBase> controller_;
607  std::thread thread_;
608  unsigned int cm_update_rate_;
609  };
